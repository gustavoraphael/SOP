	i
	#!/usr/bash
	echo digite um numero
	read n
	
	if [ $n = 0 ]; then
		echo "seu numero é zero"
	
	elif [ $n -lt 0 ]; then
		echo "seu numero é negativo"
	
	else
		echo "seu numero é positivo"

	fi

	chmod +x terceiro.sh
	sh terceiro.sh

questão 4

	
	vi quarto.sh
	i
	#!/usr/bash
	echo digite um numero 
	read n
	c=0
	echo numero $n
	while [ $c -le 10 ]; do
		val=$(($n*$c))
		echo "$n x $c = $val"
		c=$(($c+1))
	done

	chmod +x quarto.sh
	sh quarto.sh
